00508     public static Field<?> getQuickFixFieldFromName(String inFieldName)

00510     {
00511         if ( inFieldName == null ) {

00518         }
00519         catch ( ClassNotFoundException cnfe ) {
00520             // if this exception is thrown, that means that the field does not correspond to a pre-packaged quickfix field

00522             try {
00523                 int fieldInt = Integer.parseInt(inFieldName);
00524                 return new CustomField<Integer>(fieldInt,

00527             catch ( Throwable t ) {
00528                 error = t;
00529             }
00530         }
00531         catch ( Throwable t ) {
00532             error = t;
00533         }
00534         // bah, can't create a field with the stuff we're given
00535         throw new Exception(error);
00536     }
