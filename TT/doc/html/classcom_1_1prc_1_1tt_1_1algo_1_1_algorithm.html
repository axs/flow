            Order ord = new OrderBuilder()
                        .withSide( side )
                        .withOrdType( OrdType.LIMIT )
                        .withPrice( price )
                        .withOrderQty( Math.abs(currentpos) )
                        .withMsgType( MsgType.ORDER_SINGLE )
                        .withInstrument( myin )
                        .withSourceID( cfg.getSourceID() )
                        .create();

            currentstate = State.FLAT;
            sendOrder(ord);
        }
